Q7. Write a function that accepts parameter n and returns factorial of n

Ans:
      function factorial(n)
      {
         if(n===0)
         {
            console.log("1")
         }
         else{
            var fact=1;
      for(var i=n;i>0;i--)
      {
          fact=fact*i
      }
      console.log(fact)
         }
      }
      
      factorial(5)

      Output:120
